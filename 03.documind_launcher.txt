Using your documind.sh launcher
# stop backend only
./scripts/documind.sh stop service backend

# stop UI only
./scripts/documind.sh stop service ui

# stop both
./scripts/documind.sh stop service both

Using systemd directly
# backend
sudo systemctl stop documind-backend.service

# UI (whichever you use)
sudo systemctl stop frontend.service        # if your UI service is named 'frontend.service'
# or
sudo systemctl stop documind-ui.service

Prevent auto-start at boot (optional)
sudo systemctl disable documind-backend.service
sudo systemctl disable frontend.service
